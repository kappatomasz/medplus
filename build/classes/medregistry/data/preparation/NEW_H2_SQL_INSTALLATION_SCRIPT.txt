CREATE TABLE PATIENTS(
ID BIGINT PRIMARY KEY NOT NULL,
NAME VARCHAR(30),
SECOND_NAME VARCHAR(30),
SURNAME VARCHAR(50),
STREET VARCHAR(50),
HOUSE_NO VARCHAR(7),
FLAT_NO VARCHAR(7),
POST_CODE VARCHAR(5),
CITY VARCHAR(50),
STATE_ID INT(2),
PHONE VARCHAR(20),
DISEASE_ID BIGINT(11)
);

CREATE UNIQUE INDEX patientIDX ON PATIENTS(ID);

CREATE TABLE ATTACHMENTS(
ID BIGINT(11) PRIMARY KEY NOT NULL,
PATIENT_ID BIGINT(11) NOT NULL,
CONTENT_TYPE VARCHAR(20),
BINARY_DATA BLOB NOT NULL,
FILE_NAME VARCHAR(50) NOT NULL);

ALTER TABLE ATTACHMENTS ADD FOREIGN KEY(PATIENT_ID) REFERENCES PATIENTS(ID);
CREATE INDEX ATH_IDX ON ATTACHMENTS(ID);

CREATE TABLE DISEASES(
ID BIGINT PRIMARY KEY NOT NULL,
NAME VARCHAR(100) NOT NULL,
DESCRIPTION BLOB);
CREATE INDEX DIS_IDX ON DISEASES(ID);

CREATE TABLE STATES(
ID INT(2),
NAME VARCHAR(30));
